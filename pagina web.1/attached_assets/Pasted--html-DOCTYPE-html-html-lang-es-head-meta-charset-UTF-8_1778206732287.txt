function tick(){

  const est=new Date(utc-18000000);

},{threshold:.3});

function fmtText(t){
  return t
    .replace(/\*\*([^*]+)\*\*/g,'<strong>$1</strong>')
    .replace(/\*([^*]+)\*/g,'<em>$1</em>')
    .replace(/^### (.+)$/gm,'<div style="font-family:var(--fn-d);font-size:.72rem;color:var(--gold);letter-spacing:.1em;margin:14px 0 5px;font-weight:600;">$1</div>')
    .replace(/^## (.+)$/gm,'<div style="font-family:var(--fn-d);font-size:.8rem;color:var(--cyan);letter-spacing:.08em;margin:16px 0 7px;font-weight:700;">$1</div>')
    .replace(/^# (.+)$/gm,'<div style="font-family:var(--fn-d);font-size:.9rem;color:var(--snow);letter-spacing:.05em;margin:18px 0 8px;font-weight:800;">$1</div>')
    .replace(/^- (.+)$/gm,'<div style="padding-left:10px;margin:2px 0;color:var(--pale);">→ $1</div>')
    .replace(/\n\n/g,'</p><p style="margin-top:10px;">')
    .replace(/\n/g,'<br>');
}
